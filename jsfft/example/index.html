Sample data 
var data = new ComplexArray(128)
data.map(function(value, i, n) {
  value.real = (i > n/3 && i < 2*n/3) ? 1 : 0
})
Transform (in place):
data.FFT()
Simple low pass filter: 
data.map(function(freq, i, n) {
  if (i > n/5 && i < 4*n/5) {
    freq.real = 0
    freq.imag = 0
  }
})
Transform back:
data.InvFFT()
... or all in one step 
data.frequencyMap(function(freq, i, n) {
  if (i > n/5 && i < 4*n/5) {
    freq.real = 0
    freq.imag = 0
  }
})
